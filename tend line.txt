"""
Auto Trend Channels Indicator - Complete Implementation
Based on FXSSI Auto Trend Channel algorithm

Features:
- Detects multiple simultaneous channels
- Supports different timeframes
- Dynamic channel transitions on breakouts
- Projection lines for future trends
- Proper swing point detection
- Channel validation and scoring
"""

import numpy as np
import pandas as pd
from typing import List, Tuple, Optional, Dict
from dataclasses import dataclass, field
from enum import Enum
import matplotlib.pyplot as plt
import matplotlib.patches as mpatches
from matplotlib.lines import Line2D


class ChannelType(Enum):
    """Type of trend channel"""
    ASCENDING = "ascending"
    DESCENDING = "descending"
    

@dataclass
class Channel:
    """Represents a detected trend channel"""
    upper_slope: float
    upper_intercept: float
    lower_slope: float
    lower_intercept: float
    start_idx: int
    end_idx: int
    channel_type: ChannelType
    timeframe: str
    confidence: float
    is_active: bool = True  # Whether channel is currently active
    broken_at: Optional[int] = None  # Index where channel was broken
    break_direction: Optional[str] = None  # 'upper' or 'lower'
    
    def get_upper_value(self, idx: int) -> float:
        """Get upper channel value at index"""
        return self.upper_slope * idx + self.upper_intercept
    
    def get_lower_value(self, idx: int) -> float:
        """Get lower channel value at index"""
        return self.lower_slope * idx + self.lower_intercept
    
    def get_middle_value(self, idx: int) -> float:
        """Get middle line value at index"""
        return (self.get_upper_value(idx) + self.get_lower_value(idx)) / 2
    
    def get_width(self, idx: int) -> float:
        """Get channel width at index"""
        return self.get_upper_value(idx) - self.get_lower_value(idx)


class AutoTrendChannels:
    """
    Auto Trend Channels Indicator
    
    Detects and tracks multiple parallel trend channels across price data.
    Handles dynamic updates when channels are broken.
    """
    
    def __init__(
        self,
        min_history_bars: int = 50,
        swing_detection_window: int = 5,
        max_channel_angle: float = 85.0,
        min_touches: int = 2,
        touch_tolerance: float = 0.015,
        projection_bars: int = 50,
        enable_dynamic_updates: bool = True
    ):
        """
        Initialize the Auto Trend Channels indicator.
        
        Parameters:
        -----------
        min_history_bars : int
            Minimum history depth in bars required for channel formation
        swing_detection_window : int
            Window size for detecting swing highs/lows
        max_channel_angle : float
            Maximum angle in degrees for valid channels
        min_touches : int
            Minimum number of touches required for valid channel
        touch_tolerance : float
            Tolerance for price touching channel lines (as fraction of price)
        projection_bars : int
            Number of bars to project channel into the future
        enable_dynamic_updates : bool
            Whether to detect new channels when price breaks existing ones
        """
        self.min_history_bars = min_history_bars
        self.swing_detection_window = swing_detection_window
        self.max_channel_angle = max_channel_angle
        self.min_touches = min_touches
        self.touch_tolerance = touch_tolerance
        self.projection_bars = projection_bars
        self.enable_dynamic_updates = enable_dynamic_updates
        
    def find_swing_highs(self, high: np.ndarray, window: int) -> np.ndarray:
        """Find swing high points in the price data."""
        swing_highs = np.zeros(len(high), dtype=bool)
        
        for i in range(window, len(high) - window):
            is_swing_high = True
            for j in range(-window, window + 1):
                if j == 0:
                    continue
                if high[i] <= high[i + j]:
                    is_swing_high = False
                    break
            swing_highs[i] = is_swing_high
            
        return swing_highs
    
    def find_swing_lows(self, low: np.ndarray, window: int) -> np.ndarray:
        """Find swing low points in the price data."""
        swing_lows = np.zeros(len(low), dtype=bool)
        
        for i in range(window, len(low) - window):
            is_swing_low = True
            for j in range(-window, window + 1):
                if j == 0:
                    continue
                if low[i] >= low[i + j]:
                    is_swing_low = False
                    break
            swing_lows[i] = is_swing_low
            
        return swing_lows
    
    def fit_line_through_points(
        self, 
        x1: float, y1: float,
        x2: float, y2: float
    ) -> Tuple[float, float]:
        """Fit a line through two points."""
        if x1 == x2:
            return 0.0, y1
        slope = (y2 - y1) / (x2 - x1)
        intercept = y1 - slope * x1
        return slope, intercept
    
    def find_parallel_line(
        self,
        slope: float,
        indices: np.ndarray,
        values: np.ndarray,
        is_upper: bool
    ) -> float:
        """Find intercept of parallel line touching extreme points."""
        intercepts = values - slope * indices
        return np.max(intercepts) if is_upper else np.min(intercepts)
    
    def score_channel(
        self,
        upper_slope: float, upper_intercept: float,
        lower_slope: float, lower_intercept: float,
        high: np.ndarray, low: np.ndarray,
        swing_high_idx: np.ndarray, swing_high_val: np.ndarray,
        swing_low_idx: np.ndarray, swing_low_val: np.ndarray,
        start_idx: int, end_idx: int
    ) -> Tuple[float, int]:
        """Score a channel based on containment and touches."""
        # Only score within the channel's range
        indices = np.arange(start_idx, min(end_idx + 1, len(high)))
        upper_line = upper_slope * indices + upper_intercept
        lower_line = lower_slope * indices + lower_intercept
        
        # Count bars contained within channel
        high_slice = high[start_idx:min(end_idx + 1, len(high))]
        low_slice = low[start_idx:min(end_idx + 1, len(low))]
        
        contained = np.sum(
            (low_slice >= lower_line * (1 - self.touch_tolerance)) & 
            (high_slice <= upper_line * (1 + self.touch_tolerance))
        )
        
        # Count touches on boundaries
        # Filter swing points to only those in range
        valid_highs = (swing_high_idx >= start_idx) & (swing_high_idx <= end_idx)
        valid_lows = (swing_low_idx >= start_idx) & (swing_low_idx <= end_idx)
        
        filtered_high_idx = swing_high_idx[valid_highs]
        filtered_high_val = swing_high_val[valid_highs]
        filtered_low_idx = swing_low_idx[valid_lows]
        filtered_low_val = swing_low_val[valid_lows]
        
        upper_touches = 0
        if len(filtered_high_idx) > 0:
            upper_predicted = upper_slope * filtered_high_idx + upper_intercept
            upper_distances = np.abs(filtered_high_val - upper_predicted)
            upper_touches = np.sum(upper_distances <= (filtered_high_val * self.touch_tolerance))
        
        lower_touches = 0
        if len(filtered_low_idx) > 0:
            lower_predicted = lower_slope * filtered_low_idx + lower_intercept
            lower_distances = np.abs(filtered_low_val - lower_predicted)
            lower_touches = np.sum(lower_distances <= (filtered_low_val * self.touch_tolerance))
        
        total_touches = upper_touches + lower_touches
        
        # Calculate score
        if len(indices) == 0:
            containment_ratio = 0
        else:
            containment_ratio = contained / len(indices)
        
        score = containment_ratio * 100 + total_touches * 20
        
        return score, total_touches
    
    def detect_channel_in_range(
        self,
        high: np.ndarray,
        low: np.ndarray,
        start_idx: int,
        end_idx: int,
        timeframe: str = "current"
    ) -> Optional[Channel]:
        """
        Detect a single channel within a specific range.
        """
        if end_idx - start_idx < self.min_history_bars:
            return None
        
        # Find swing points in the range
        swing_highs = self.find_swing_highs(high, self.swing_detection_window)
        swing_lows = self.find_swing_lows(low, self.swing_detection_window)
        
        swing_high_indices = np.where(swing_highs)[0]
        swing_low_indices = np.where(swing_lows)[0]
        
        # Filter to range
        swing_high_indices = swing_high_indices[
            (swing_high_indices >= start_idx) & (swing_high_indices <= end_idx)
        ]
        swing_low_indices = swing_low_indices[
            (swing_low_indices >= start_idx) & (swing_low_indices <= end_idx)
        ]
        
        if len(swing_high_indices) < 2 or len(swing_low_indices) < 2:
            return None
        
        swing_high_values = high[swing_high_indices]
        swing_low_values = low[swing_low_indices]
        
        best_score = -np.inf
        best_channel = None
        
        # Limit combinations for performance
        max_points = 10
        recent_lows_idx = swing_low_indices[-max_points:] if len(swing_low_indices) > max_points else swing_low_indices
        recent_lows_val = swing_low_values[-max_points:] if len(swing_low_values) > max_points else swing_low_values
        recent_highs_idx = swing_high_indices[-max_points:] if len(swing_high_indices) > max_points else swing_high_indices
        recent_highs_val = swing_high_values[-max_points:] if len(swing_high_values) > max_points else swing_high_values
        
        # Try ascending channels (draw through lows)
        for i in range(len(recent_lows_idx) - 1):
            for j in range(i + 1, len(recent_lows_idx)):
                slope, intercept = self.fit_line_through_points(
                    recent_lows_idx[i], recent_lows_val[i],
                    recent_lows_idx[j], recent_lows_val[j]
                )
                
                if slope <= 0 or abs(np.degrees(np.arctan(slope))) > self.max_channel_angle:
                    continue
                
                upper_intercept = self.find_parallel_line(
                    slope, swing_high_indices, swing_high_values, is_upper=True
                )
                
                score, touches = self.score_channel(
                    slope, upper_intercept, slope, intercept,
                    high, low,
                    swing_high_indices, swing_high_values,
                    swing_low_indices, swing_low_values,
                    start_idx, end_idx
                )
                
                score += abs(slope) * 10  # Bonus for strong trends
                
                if touches >= self.min_touches and score > best_score:
                    best_score = score
                    best_channel = Channel(
                        upper_slope=slope,
                        upper_intercept=upper_intercept,
                        lower_slope=slope,
                        lower_intercept=intercept,
                        start_idx=start_idx,
                        end_idx=end_idx,
                        channel_type=ChannelType.ASCENDING,
                        timeframe=timeframe,
                        confidence=touches / (len(swing_high_indices) + len(swing_low_indices))
                    )
        
        # Try descending channels (draw through highs)
        for i in range(len(recent_highs_idx) - 1):
            for j in range(i + 1, len(recent_highs_idx)):
                slope, intercept = self.fit_line_through_points(
                    recent_highs_idx[i], recent_highs_val[i],
                    recent_highs_idx[j], recent_highs_val[j]
                )
                
                if slope >= 0 or abs(np.degrees(np.arctan(slope))) > self.max_channel_angle:
                    continue
                
                lower_intercept = self.find_parallel_line(
                    slope, swing_low_indices, swing_low_values, is_upper=False
                )
                
                score, touches = self.score_channel(
                    slope, intercept, slope, lower_intercept,
                    high, low,
                    swing_high_indices, swing_high_values,
                    swing_low_indices, swing_low_values,
                    start_idx, end_idx
                )
                
                score += abs(slope) * 10  # Bonus for strong trends
                
                if touches >= self.min_touches and score > best_score:
                    best_score = score
                    best_channel = Channel(
                        upper_slope=slope,
                        upper_intercept=intercept,
                        lower_slope=slope,
                        lower_intercept=lower_intercept,
                        start_idx=start_idx,
                        end_idx=end_idx,
                        channel_type=ChannelType.DESCENDING,
                        timeframe=timeframe,
                        confidence=touches / (len(swing_high_indices) + len(swing_low_indices))
                    )
        
        return best_channel
    
    def check_breakout(
        self,
        channel: Channel,
        high: np.ndarray,
        low: np.ndarray,
        idx: int
    ) -> Tuple[bool, Optional[str]]:
        """Check if price has broken out of the channel."""
        if idx >= len(high) or idx >= len(low):
            return False, None
        
        upper_value = channel.get_upper_value(idx)
        lower_value = channel.get_lower_value(idx)
        
        # Check for breakout
        if high[idx] > upper_value * (1 + self.touch_tolerance):
            return True, 'upper'
        elif low[idx] < lower_value * (1 - self.touch_tolerance):
            return True, 'lower'
        
        return False, None
    
    def detect_all_channels(
        self,
        high: np.ndarray,
        low: np.ndarray,
        timeframe: str = "current"
    ) -> List[Channel]:
        """
        Detect all channels in the price data with dynamic updates.
        
        This method:
        1. Detects the current active channel
        2. Monitors for breakouts
        3. When a breakout occurs, keeps the old channel and detects a new one
        """
        channels = []
        search_start = 0
        
        while search_start < len(high) - self.min_history_bars:
            # Detect channel from current start point to end
            channel = self.detect_channel_in_range(
                high, low,
                search_start, len(high) - 1,
                timeframe
            )
            
            if channel is None:
                # No channel found, move forward
                search_start += self.min_history_bars // 2
                continue
            
            # Check if this channel gets broken before the end
            if self.enable_dynamic_updates:
                for idx in range(channel.start_idx + self.min_history_bars, len(high)):
                    is_broken, direction = self.check_breakout(channel, high, low, idx)
                    
                    if is_broken:
                        # Channel broken - update it and continue searching
                        channel.end_idx = idx
                        channel.is_active = False
                        channel.broken_at = idx
                        channel.break_direction = direction
                        channels.append(channel)
                        
                        # Start searching for next channel after the breakout
                        search_start = idx
                        break
                else:
                    # Channel extends to the end
                    channels.append(channel)
                    break
            else:
                channels.append(channel)
                break
        
        return channels
    
    def fit(
        self,
        df: pd.DataFrame,
        high_col: str = 'high',
        low_col: str = 'low',
        timeframe: str = "current"
    ) -> List[Channel]:
        """
        Detect channels in a pandas DataFrame.
        """
        high = df[high_col].values
        low = df[low_col].values
        
        channels = self.detect_all_channels(high, low, timeframe)
        
        return channels
    
    def plot_channels(
        self,
        df: pd.DataFrame,
        channels: List[Channel],
        high_col: str = 'high',
        low_col: str = 'low',
        close_col: str = 'close',
        show_projections: bool = True,
        show_middle_lines: bool = True,
        figsize: Tuple[int, int] = (16, 9)
    ):
        """
        Plot price data with detected channels.
        """
        fig, ax = plt.subplots(figsize=figsize, facecolor='white')
        
        # Plot price data
        ax.plot(df.index, df[close_col], color='black', linewidth=1, label='Close', zorder=2)
        ax.fill_between(df.index, df[low_col], df[high_col], alpha=0.2, color='gray', zorder=1)
        
        # Color scheme
        colors = {
            ChannelType.ASCENDING: '#2ecc71',  # Green
            ChannelType.DESCENDING: '#e74c3c'  # Red
        }
        
        # Plot each channel
        for i, channel in enumerate(channels):
            color = colors[channel.channel_type]
            alpha = 0.8 if channel.is_active else 0.4
            linewidth = 2 if channel.is_active else 1.5
            
            # Historical part (solid lines)
            hist_indices = np.arange(channel.start_idx, channel.end_idx + 1)
            upper_hist = [channel.get_upper_value(idx) for idx in hist_indices]
            lower_hist = [channel.get_lower_value(idx) for idx in hist_indices]
            
            ax.plot(hist_indices, upper_hist, color=color, linewidth=linewidth,
                   alpha=alpha, linestyle='-', zorder=3)
            ax.plot(hist_indices, lower_hist, color=color, linewidth=linewidth,
                   alpha=alpha, linestyle='-', zorder=3)
            
            # Fill between channel lines
            ax.fill_between(hist_indices, lower_hist, upper_hist,
                           alpha=0.1, color=color, zorder=0)
            
            # Projection (dashed lines)
            if show_projections and channel.is_active:
                proj_start = channel.end_idx
                proj_indices = np.arange(proj_start, proj_start + self.projection_bars)
                upper_proj = [channel.get_upper_value(idx) for idx in proj_indices]
                lower_proj = [channel.get_lower_value(idx) for idx in proj_indices]
                
                ax.plot(proj_indices, upper_proj, color=color, linewidth=linewidth + 0.5,
                       alpha=alpha * 0.7, linestyle='--', zorder=3)
                ax.plot(proj_indices, lower_proj, color=color, linewidth=linewidth + 0.5,
                       alpha=alpha * 0.7, linestyle='--', zorder=3)
            
            # Middle line (dotted)
            if show_middle_lines:
                middle_hist = [(u + l) / 2 for u, l in zip(upper_hist, lower_hist)]
                ax.plot(hist_indices, middle_hist, color=color, linewidth=1,
                       alpha=alpha * 0.6, linestyle=':', zorder=2)
            
            # Add channel label
            label_idx = channel.start_idx + (channel.end_idx - channel.start_idx) // 4
            label_y = channel.get_upper_value(label_idx) * 1.002
            
            status = "●" if channel.is_active else "○"
            label_text = f"{status} {channel.timeframe}"
            
            ax.text(label_idx, label_y, label_text,
                   fontsize=10, fontweight='bold',
                   color=color, alpha=alpha,
                   bbox=dict(boxstyle='round,pad=0.3', facecolor='white',
                           edgecolor=color, alpha=0.7))
        
        # Mark swing points
        swing_highs = self.find_swing_highs(df[high_col].values, self.swing_detection_window)
        swing_lows = self.find_swing_lows(df[low_col].values, self.swing_detection_window)
        
        swing_high_idx = np.where(swing_highs)[0]
        swing_low_idx = np.where(swing_lows)[0]
        
        ax.scatter(swing_high_idx, df.loc[swing_high_idx, high_col],
                  color='green', marker='^', s=40, alpha=0.5, zorder=4, label='Swing Highs')
        ax.scatter(swing_low_idx, df.loc[swing_low_idx, low_col],
                  color='red', marker='v', s=40, alpha=0.5, zorder=4, label='Swing Lows')
        
        ax.set_title('Auto Trend Channels', fontsize=16, fontweight='bold')
        ax.set_xlabel('Bar Index', fontsize=12)
        ax.set_ylabel('Price', fontsize=12)
        ax.legend(loc='best')
        ax.grid(True, alpha=0.3, linestyle='--')
        
        plt.tight_layout()
        return fig, ax


def example_usage():
    """Example usage with realistic channel patterns."""
    np.random.seed(42)
    n_bars = 400
    
    # Create a price series with channel breakouts and new channels
    price = np.zeros(n_bars)
    price[0] = 100
    
    # Phase 1: Uptrend channel (bars 0-150)
    for i in range(1, 150):
        trend = 0.15
        oscillation = 3 * np.sin(i * 0.1)
        noise = np.random.randn() * 0.5
        price[i] = price[i-1] + trend + oscillation * 0.05 + noise
    
    # Phase 2: Breakout and downtrend (bars 150-280)
    for i in range(150, 280):
        trend = -0.12
        oscillation = 2.5 * np.sin(i * 0.08)
        noise = np.random.randn() * 0.5
        price[i] = price[i-1] + trend + oscillation * 0.05 + noise
    
    # Phase 3: New uptrend channel (bars 280-400)
    for i in range(280, n_bars):
        trend = 0.18
        oscillation = 3.5 * np.sin(i * 0.12)
        noise = np.random.randn() * 0.5
        price[i] = price[i-1] + trend + oscillation * 0.05 + noise
    
    # Create OHLC data
    close = price
    high = close + np.abs(np.random.randn(n_bars) * 1.2)
    low = close - np.abs(np.random.randn(n_bars) * 1.2)
    open_price = close + np.random.randn(n_bars) * 0.5
    
    # Ensure high is highest and low is lowest
    high = np.maximum(high, np.maximum(open_price, close))
    low = np.minimum(low, np.minimum(open_price, close))
    
    df = pd.DataFrame({
        'open': open_price,
        'high': high,
        'low': low,
        'close': close
    })
    
    # Initialize indicator with dynamic updates
    atc = AutoTrendChannels(
        min_history_bars=30,
        swing_detection_window=5,
        min_touches=2,
        touch_tolerance=0.02,
        projection_bars=30,
        enable_dynamic_updates=True
    )
    
    # Detect channels
    channels = atc.fit(df, timeframe="H4")
    
    print(f"Detected {len(channels)} channels\n")
    
    for i, ch in enumerate(channels, 1):
        print(f"Channel {i}:")
        print(f"  Type: {ch.channel_type.value}")
        print(f"  Timeframe: {ch.timeframe}")
        print(f"  Range: bars {ch.start_idx} to {ch.end_idx}")
        print(f"  Slope: {ch.upper_slope:.6f}")
        print(f"  Active: {ch.is_active}")
        if not ch.is_active:
            print(f"  Broken at: bar {ch.broken_at} ({ch.break_direction} breakout)")
        print(f"  Confidence: {ch.confidence:.2f}")
        print()
    
    # Plot
    fig, ax = atc.plot_channels(df, channels, show_projections=True, show_middle_lines=True)
    plt.savefig('/home/claude/auto_channels_multi.png', dpi=150, bbox_inches='tight')
    print("Plot saved to 'auto_channels_multi.png'")
    
    return df, channels, atc


if __name__ == "__main__":
    example_usage()
